Solution

try:
    return feedparser.parse(url)
except TypeError:
    if 'drv_libxml2' in feedparser.PREFERRED_XML_PARSERS:
        feedparser.PREFERRED_XML_PARSERS.remove('drv_libxml2')
        return feedparser.parse(url)
    else:
        return None
